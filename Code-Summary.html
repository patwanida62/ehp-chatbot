require('dotenv').config();
const express = require('express');
const crypto  = require('crypto');
const axios   = require('axios');

const app = express();

// ── CONFIG ────────────────────────────────────────────────────
const LINE_CHANNEL_ACCESS_TOKEN = process.env.LINE_CHANNEL_ACCESS_TOKEN;
const LINE_CHANNEL_SECRET       = process.env.LINE_CHANNEL_SECRET;
const TICKET_API_BASE_URL       = process.env.TICKET_API_BASE_URL || 'http://localhost:4000';
const PORT                      = process.env.PORT || 3000;

// ── MIDDLEWARE ────────────────────────────────────────────────
app.use(express.json({ verify: (req, _res, buf) => { req.rawBody = buf; } }));

// ── HELPER: Verify LINE Signature ────────────────────────────
function verifyLineSignature(req) {
  const signature = req.headers['x-line-signature'];
  if (!signature) return false;
  const hash = crypto.createHmac('SHA256', LINE_CHANNEL_SECRET)
    .update(req.rawBody).digest('base64');
  return hash === signature;
}

// ── HELPER: ส่ง Push Message ─────────────────────────────────
async function sendLineMessage(userId, messages) {
  await axios.post('https://api.line.me/v2/bot/message/push',
    { to: userId, messages },
    { headers: { Authorization: `Bearer ${LINE_CHANNEL_ACCESS_TOKEN}` } }
  );
}

// ── HELPER: ดึงชื่อ LINE Display Name ────────────────────────
async function getLineDisplayName(userId) {
  try {
    const res = await axios.get(`https://api.line.me/v2/bot/profile/${userId}`,
      { headers: { Authorization: `Bearer ${LINE_CHANNEL_ACCESS_TOKEN}` } }
    );
    return res.data.displayName;
  } catch { return 'ไม่ทราบชื่อ'; }
}

// ── ROUTE 1: Webhook จาก LINE ────────────────────────────────
app.post('/webhook/line', async (req, res) => {
  if (!verifyLineSignature(req)) return res.status(401).json({ error: 'Invalid signature' });
  res.status(200).end(); // LINE ต้องการ 200 OK เร็วๆ

  for (const event of req.body.events || []) {
    if (event.type !== 'message' || event.message.type !== 'text') continue;
    const userId = event.source.userId;
    const text   = event.message.text;

    // โหลด Keyword จาก API
    let keywords = [];
    try { keywords = (await axios.get(`${TICKET_API_BASE_URL}/keywords`)).data; }
    catch {}

    // ตรวจ keyword → เปิด Ticket อัตโนมัติ
    const matched = keywords.find(kw => text.toLowerCase().includes(kw.toLowerCase()));
    if (matched) {
      const [displayName, settings] = await Promise.all([
        getLineDisplayName(userId),
        axios.get(`${TICKET_API_BASE_URL}/settings`).then(r => r.data),
      ]);
      const ticket = (await axios.post(`${TICKET_API_BASE_URL}/tickets`, {
        title: text, lineUserId: userId, reporterName: displayName,
        service: settings.service, category: settings.category,
        subCategory: settings.subCategory,
      })).data;
      await sendLineMessage(userId, [{ type: 'text', text:
        `📋 Ticket: #${ticket.id}\n🔴 ปัญหา: ${ticket.title}\n👤 ผู้แจ้ง: ${ticket.reporterName}\n` +
        `🏢 Service: ${ticket.service}\n📂 Category: ${ticket.category}\n` +
        `📌 Sub Category: ${ticket.subCategory}\n⚙️ สถานะ: ${settings.defaultStatus}\n` +
        `🔧 วิธีการแก้ไข: -\n📅 รับเรื่องวันที่: ${new Date(ticket.createdAt).toLocaleString('th-TH')}\n` +
        `\nติดตามสถานะได้โดยพิมพ์:\n"ตรวจสอบ #${ticket.id}"`
      }]);
      continue;
    }

    // ตรวจสอบ #ID → แสดงสถานะ Ticket
    const trackMatch = text.match(/ตรวจสอบ\s*#?(\d+)/);
    if (trackMatch) {
      const ticket = (await axios.get(`${TICKET_API_BASE_URL}/tickets/${trackMatch[1]}`)).data;
      const isClosed = ticket.status === 'closed';
      await sendLineMessage(userId, [{ type: 'text', text:
        `📋 สถานะ Ticket #${ticket.id}\n🔴 ปัญหา: ${ticket.title}\n` +
        `⚙️ สถานะ: ${translateStatus(ticket.status)}\n` +
        `🔧 วิธีการแก้ไข: ${ticket.resolution || '-'}\n` +
        `👨‍💻 ผู้ดำเนินการแก้ไข: ${ticket.resolvedBy || '-'}\n` +
        (isClosed ? `✅ วันที่แก้ไขเสร็จ: ${new Date(ticket.resolvedAt).toLocaleString('th-TH')}\n`
                  : `🕐 อัปเดตล่าสุด: ${new Date(ticket.updatedAt).toLocaleString('th-TH')}\n`)
      }]);
      continue;
    }
  }
});

// ── ROUTE 2: แจ้งเตือนเปิด Ticket ──────────────────────────
app.post('/notify/ticket-opened', async (req, res) => {
  const { ticketId, title, lineUserId } = req.body;
  await sendLineMessage(lineUserId, [{ type: 'text', text:
    `✅ เปิด Ticket #${ticketId} เรียบร้อยแล้วครับ\n📌 หัวข้อ: ${title}\n` +
    `ติดตามสถานะ: "ตรวจสอบ #${ticketId}"`
  }]);
  res.json({ success: true });
});

// ── ROUTE 3: แจ้งเตือนปิด Ticket ──────────────────────────
app.post('/notify/ticket-closed', async (req, res) => {
  const { ticketId, title, lineUserId } = req.body;
  await sendLineMessage(lineUserId, [{ type: 'text', text:
    `🎉 Ticket #${ticketId} ดำเนินการเสร็จสิ้นแล้วครับ\n📌 หัวข้อ: ${title}\n` +
    `หากยังมีปัญหา สามารถแจ้งกลับมาได้เลยครับ`
  }]);
  res.json({ success: true });
});

// ── HELPER: แปลงสถานะเป็นภาษาไทย ─────────────────────────
function translateStatus(status) {
  return { open: '🟡 เปิดแล้ว รอดำเนินการ', in_progress: '🔵 กำลังดำเนินการ',
           closed: '🟢 ดำเนินการเสร็จสิ้น', cancelled: '🔴 ยกเลิกแล้ว' }[status] || status;
}

app.listen(PORT, () => console.log(`Webhook server running on port ${PORT}`));

require('dotenv').config();
const express = require('express');
const fs      = require('fs');
const path    = require('path');
const app = express();
app.use(express.json());

// ── File helpers ──────────────────────────────────────────────
const KEYWORDS_FILE = path.join(__dirname, 'keywords.json');
const SETTINGS_FILE = path.join(__dirname, 'settings.json');

const loadKeywords  = () => JSON.parse(fs.readFileSync(KEYWORDS_FILE, 'utf-8'));
const saveKeywords  = kw => fs.writeFileSync(KEYWORDS_FILE, JSON.stringify(kw, null, 2));
const loadSettings  = () => JSON.parse(fs.readFileSync(SETTINGS_FILE, 'utf-8'));
const saveSettings  = s  => fs.writeFileSync(SETTINGS_FILE, JSON.stringify(s, null, 2));

// ── Keywords Routes ───────────────────────────────────────────
app.get('/keywords', (req, res) => res.json(loadKeywords()));
app.post('/keywords', (req, res) => {
  const keywords = loadKeywords();
  const kw = req.body.keyword?.trim();
  if (!kw) return res.status(400).json({ error: 'keyword is required' });
  if (keywords.includes(kw)) return res.status(400).json({ error: 'Keyword นี้มีอยู่แล้ว' });
  keywords.push(kw); saveKeywords(keywords);
  res.status(201).json({ success: true, keywords });
});
app.delete('/keywords/:index', (req, res) => {
  const keywords = loadKeywords();
  keywords.splice(parseInt(req.params.index), 1);
  saveKeywords(keywords);
  res.json({ success: true, keywords });
});

// ── Settings Routes ───────────────────────────────────────────
app.get('/settings', (req, res) => res.json(loadSettings()));
app.put('/settings', (req, res) => {
  const updated = { ...loadSettings(), ...req.body };
  saveSettings(updated);
  res.json(updated);
});

// ── Admin UI ──────────────────────────────────────────────────
app.get('/admin', (req, res) => res.sendFile(path.join(__dirname, 'admin.html')));

// ── In-memory Ticket Storage ──────────────────────────────────
let tickets = [
  { id: '1', title: 'ปัญหาการ Login ไม่ได้', status: 'open', resolution: '', resolvedBy: '', resolvedAt: null },
  { id: '2', title: 'ระบบช้ามากช่วงเช้า',    status: 'in_progress', resolution: '', resolvedBy: '', resolvedAt: null },
];
let nextId = 3;

// ── Ticket Routes ─────────────────────────────────────────────
app.get('/tickets', (req, res) => res.json(tickets));

app.get('/tickets/:id', (req, res) => {
  const t = tickets.find(t => t.id === req.params.id);
  t ? res.json(t) : res.status(404).json({ error: 'Ticket not found' });
});

app.post('/tickets', (req, res) => {
  const { title, lineUserId, reporterName, service, category, subCategory } = req.body;
  if (!title || !lineUserId) return res.status(400).json({ error: 'title and lineUserId are required' });
  const ticket = {
    id: String(nextId++), title, status: 'in_progress',
    lineUserId, reporterName: reporterName || 'ไม่ทราบชื่อ',
    service: service || 'EHP CIS', category: category || 'ปัญหาการใช้งานทั่วไป',
    subCategory: subCategory || 'การใช้งานทั่วไป',
    resolution: '', resolvedBy: '', resolvedAt: null,
    createdAt: new Date().toISOString(), updatedAt: new Date().toISOString(),
  };
  tickets.push(ticket);
  res.status(201).json(ticket);
});

app.patch('/tickets/:id/status', (req, res) => {
  const ticket = tickets.find(t => t.id === req.params.id);
  if (!ticket) return res.status(404).json({ error: 'Ticket not found' });
  const { status, resolution, resolvedBy } = req.body;
  ticket.status    = status;
  ticket.updatedAt = new Date().toISOString();
  if (resolution !== undefined) ticket.resolution = resolution;
  if (resolvedBy !== undefined) ticket.resolvedBy  = resolvedBy;
  if (status === 'closed') ticket.resolvedAt = new Date().toISOString();
  res.json(ticket);
});

app.delete('/tickets/:id', (req, res) => {
  const i = tickets.findIndex(t => t.id === req.params.id);
  if (i === -1) return res.status(404).json({ error: 'Ticket not found' });
  tickets.splice(i, 1);
  res.json({ success: true });
});

const PORT = process.env.TICKET_API_PORT || 4000;
app.listen(PORT, () => console.log(`Ticket API running on port ${PORT}`));
